À propos du projet

Bitmap Image Processing est une bibliothèque C# avancée pour le traitement et la manipulation d'images au format BMP. Ce projet a été développé pour explorer les principes fondamentaux du traitement d'images numériques et des algorithmes de compression, en travaillant directement avec les données binaires des fichiers BMP sans dépendre de bibliothèques externes de traitement d'images.

La bibliothèque offre non seulement des fonctionnalités de manipulation d'images (filtres, transformations, etc.), mais également une implémentation complète de l'algorithme de compression de Huffman, permettant de réduire significativement la taille des fichiers tout en conservant l'intégralité des données d'image.

Fonctionnalités principales

  • Lecture et écriture de fichiers BMP avec prise en charge de différentes profondeurs de couleur
  • Filtres d'images (noir et blanc, négatif, sépia, etc.)
  • Opérations de transformation (rotation, redimensionnement, miroir)
  • Compression d'image utilisant l'algorithme de Huffman avec encodage et décodage
  • Extraction et manipulation des canaux de couleur (rouge, vert, bleu)
  • Histogrammes de couleur et analyse d'image
  • Interface graphique pour prévisualiser et appliquer les transformations
  • Exportation vers différents formats (BMP original, format compressé personnalisé)

Structure d'un fichier BMP


public class BitmapFile
{
    // En-tête du fichier BMP (14 bytes)
    public struct BITMAPFILEHEADER
    {
        public ushort bfType;      // Type de fichier ("BM")
        public uint bfSize;        // Taille du fichier en bytes
        public ushort bfReserved1; // Réservé, doit être 0
        public ushort bfReserved2; // Réservé, doit être 0
        public uint bfOffBits;     // Offset des données de l'image depuis le début du fichier
    }

    // En-tête d'information BMP (40 bytes pour BITMAPINFOHEADER)
    public struct BITMAPINFOHEADER
    {
        public uint biSize;          // Taille de cette structure en bytes
        public int biWidth;          // Largeur de l'image en pixels
        public int biHeight;         // Hauteur de l'image en pixels
        public ushort biPlanes;      // Nombre de plans, doit être 1
        public ushort biBitCount;    // Bits par pixel (1, 4, 8, 16, 24, ou 32)
        public uint biCompression;   // Type de compression (0 = non compressé)
        public uint biSizeImage;     // Taille des données de l'image en bytes
        public int biXPelsPerMeter;  // Résolution horizontale en pixels par mètre
        public int biYPelsPerMeter;  // Résolution verticale en pixels par mètre
        public uint biClrUsed;       // Nombre de couleurs utilisées dans la palette
        public uint biClrImportant;  // Nombre de couleurs importantes (0 = toutes)
    }

    // Table des couleurs (palette) utilisée pour les images avec moins de 24 bits par pixel
    public struct RGBQUAD
    {
        public byte rgbBlue;      // Composante bleue
        public byte rgbGreen;     // Composante verte
        public byte rgbRed;       // Composante rouge
        public byte rgbReserved;  // Réservé, doit être 0
    }

    public BITMAPINFOHEADER infoHeader;
    public RGBQUAD[] colorTable;  // Palette de couleurs si nécessaire
    public byte[] imageData;      // Données de l'image

Implémentation de l'algorithme de compression de Huffman


public class HuffmanCompression
{
    // Nœud de l'arbre de Huffman
    private class HuffmanNode : IComparable
    {
        public byte Value { get; set; }         // Valeur du byte (seulement pour les feuilles)
        public int Frequency { get; set; }      // Fréquence d'apparition
        public HuffmanNode Left { get; set; }   // Nœud enfant gauche
        public HuffmanNode Right { get; set; }  // Nœud enfant droit
        public bool IsLeaf => Left == null && Right == null;

        // Comparaison basée sur la fréquence pour la priorité dans la file
        public int CompareTo(HuffmanNode other)
        {
            return Frequency.CompareTo(other.Frequency);
        }
    }

    // Compresse des données en utilisant l'algorithme de Huffman
    public byte[] Compress(byte[] data)
    {
        if (data == null || data.Length == 0)
            return new byte[0];

        // 1. Calculer les fréquences de chaque byte
        Dictionary frequencies = new Dictionary();
        foreach (byte b in data)
        {
            if (frequencies.ContainsKey(b))
                frequencies[b]++;
            else
                frequencies[b] = 1;
        }

        // 2. Construire l'arbre de Huffman
        HuffmanNode root = BuildHuffmanTree(frequencies);

        // 3. Générer les codes de Huffman pour chaque byte
        Dictionary huffmanCodes = new Dictionary();
        GenerateHuffmanCodes(root, "", huffmanCodes);

        // 4. Encoder les données avec les codes de Huffman
        StringBuilder encodedData = new StringBuilder();
        foreach (byte b in data)
        {
            encodedData.Append(huffmanCodes[b]);
        }

        // 5. Convertir la chaîne binaire en tableau de bytes
        string binaryString = encodedData.ToString();
        byte[] compressedData = ConvertBinaryStringToByteArray(binaryString);

        // 6. Créer l'en-tête avec les informations nécessaires à la décompression
        byte[] header = CreateHeader(frequencies, data.Length);

        // 7. Combiner l'en-tête et les données compressées
        byte[] result = new byte[header.Length + compressedData.Length];
        Buffer.BlockCopy(header, 0, result, 0, header.Length);
        Buffer.BlockCopy(compressedData, 0, result, header.Length, compressedData.Length);

        return result;
    }

    // Construction de l'arbre de Huffman à partir des fréquences
    private HuffmanNode BuildHuffmanTree(Dictionary frequencies)
    {
        // Utilisation d'une file de priorité (min-heap)
        PriorityQueue priorityQueue = new PriorityQueue();

        // Ajouter tous les bytes comme feuilles dans la file de priorité
        foreach (var pair in frequencies)
        {
            priorityQueue.Enqueue(new HuffmanNode
            {
                Value = pair.Key,
                Frequency = pair.Value
            });
        }

        // Construire l'arbre en combinant les nœuds de plus basse fréquence
        while (priorityQueue.Count > 1)
        {
            HuffmanNode left = priorityQueue.Dequeue();
            HuffmanNode right = priorityQueue.Dequeue();

            HuffmanNode parent = new HuffmanNode
            {
                Frequency = left.Frequency + right.Frequency,
                Left = left,
                Right = right
            };

            priorityQueue.Enqueue(parent);
        }

        // Le dernier nœud restant est la racine de l'arbre
        return priorityQueue.Dequeue();
    }

    // Génération récursive des codes de Huffman
    private void GenerateHuffmanCodes(HuffmanNode node, string code, Dictionary huffmanCodes)
    {
        if (node == null)
            return;

        // Si c'est une feuille, on a trouvé un code complet
        if (node.IsLeaf)
        {
            huffmanCodes[node.Value] = code.Length > 0 ? code : "0"; // Cas spécial pour un seul byte unique
            return;
        }

        // Parcours récursif (gauche = 0, droite = 1)
        GenerateHuffmanCodes(node.Left, code + "0", huffmanCodes);
        GenerateHuffmanCodes(node.Right, code + "1", huffmanCodes);
    }

Défis techniques

  1. Manipulation de fichiers BMP au niveau binaire

    Le défi principal était de comprendre et manipuler le format BMP au niveau binaire, en tenant compte des spécificités du format comme le padding des lignes et les différentes structures d'en-tête.

  2. Implémentation efficace de l'algorithme de Huffman

    Développer une implémentation efficace de l'algorithme de compression de Huffman, avec la construction de l'arbre, la génération des codes et la gestion des bits individuels.

  3. Gestion de la mémoire pour les grandes images

    Optimiser l'utilisation de la mémoire pour traiter des images de grande taille, en particulier lors des opérations de transformation qui nécessitent des copies temporaires des données.

  4. Conception d'une interface utilisateur intuitive

    Créer une interface utilisateur qui permet de visualiser et manipuler facilement les images tout en présentant des informations techniques pertinentes sur le format et la compression.

Résultats et apprentissages

Ce projet m'a permis d'approfondir mes connaissances en programmation C# et en traitement d'images numériques :

  • Compréhension approfondie du format de fichier BMP et des principes des formats d'image en général
  • Maîtrise des techniques de manipulation de données binaires en C#
  • Implémentation et optimisation d'algorithmes de compression sans perte
  • Développement d'interfaces utilisateur pour la visualisation et l'édition d'images
  • Gestion efficace de la mémoire pour le traitement de grands volumes de données

La bibliothèque peut traiter efficacement des images BMP de différentes profondeurs de couleur et offre des taux de compression significatifs grâce à l'implémentation optimisée de l'algorithme de Huffman. L'interface utilisateur permet d'appliquer facilement les différentes transformations et de visualiser les résultats en temps réel.

Utilisation de la bibliothèque

  1. Chargement d'une image BMP

    
BitmapProcessor processor = new BitmapProcessor();
BitmapFile bmpFile = processor.LoadBitmapFile("image.bmp");

  2. Application d'un filtre

    
// Conversion en niveaux de gris
processor.ApplyGrayscaleFilter(bmpFile);

// Application d'un filtre sépia
processor.ApplySepiaFilter(bmpFile);

// Inversion des couleurs
processor.ApplyNegativeFilter(bmpFile);

  3. Transformation d'image

    
// Rotation de 90 degrés
BitmapFile rotatedImage = processor.RotateImage(bmpFile, 90);

// Redimensionnement
BitmapFile resizedImage = processor.ResizeImage(bmpFile, 800, 600);

// Effet miroir horizontal
BitmapFile mirroredImage = processor.FlipHorizontal(bmpFile);

  4. Compression et sauvegarde

    
// Compression avec l'algorithme de Huffman
HuffmanCompression huffman = new HuffmanCompression();
byte[] compressedData = huffman.Compress(bmpFile.imageData);

// Sauvegarde de l'image compressée
processor.SaveCompressedImage(bmpFile, compressedData, "image_compressed.hbmp");

// Sauvegarde d'une image BMP modifiée
processor.SaveBitmapFile(bmpFile, "image_modified.bmp");
